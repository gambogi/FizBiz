\begin{problem}{\problemnum \, \textsf{(POOLE 1.1.18)}}
Solve for the vector $\bx$ in terms of the vector $\ba$ and $\bc$.\\
\[\bx + 2\ba - \bb = 3(\bx + \ba) - 2(2\ba - \bb)\]
\end{problem}
\begin{solution}
	\begin{align*}
		\bx + 2\ba - \bb &= 3(\bx + \ba) - 2(2\ba - \bb) \\
		\bx + 2\ba - \bb &= 3\bx + 3\ba - 4\ba + 2\bb \\
		           -2\bx &= -3\ba - 3\bb \\
		             \bx &= {-3(\bb + \ba) \over 2}
	\end{align*}
\end{solution}
